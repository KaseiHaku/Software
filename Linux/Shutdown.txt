shell> shutdown -h now                                             # 立马关闭电源
shell> shutdown -r +10                                             # 10 分钟后重启
shell> shutdown -k 23:50 [warning info]                            # 模拟关机（不是真的关机），只是向登录者发送警告信息出去！
shell> shutdown -c                                                 # 取消关机任务
