Character {
    h               # 左    
    j               # 下
    k               # 上
    l               # 右
    t               # tx    # 下一个字符 x 的前一个字符上
    T               # Tx    # 上一个字符 x 的后一个字符上
    f               # fx    # 下一个字符 x 上  
    F               # Fx    # 上一个字符 x 上
    ;               # 重复执行上一次 f F t T 命令
    ,               # 反方向重复执行上一次 f F t T 命令
}

Word {
    b               # 当前 或 上一个 单词的词首
    e               # 当前 或 下一个 单词的词尾 
    w               # 下一个单词的词首（单词指：由 vimrc 中 iskeyword 指定的字符组成的字符串）
    ge              # 上一个单词的词尾
    
}

String {
    B               # 当前 或 上一个 字符串首
    E               # 当前 或 下一个 字符串尾
    W               # 下一个字符串首（字符串指：不包含任何空白符，即已空白符作为分界点） 
    gE              # 上一个字符串尾
    

    (               # 句首
    )               # 下一句首
}

Row {
    :67                                 # 光标移动到 67 行     
    $               # 行尾
    ^               # 行首个非空白字符
    0               # 行首
    -               # 前一行首
    +               # 后一行首    
    |               # 行首
    H               # 屏幕顶行
    M               # 屏幕中间行
    L               # 屏幕底行
}

Chunk {
    %               # 跳转到匹配的括号处

}

Paragraph {
    [               # [[    # 文件首
    {               # 段首
    ]               # ]]    # 文件尾
    }               # 段尾
}

Page {
    Ctrl+E                          # 上滚
    Ctrl+Y                          # 下滚
        
    Ctrl+D                          # 向下翻半页
    Ctrl+U                          # 向上翻半页
    
    Ctrl+F                          # 下一页
    Ctrl+B                          # 上一页
    
    9Ctrl+F                         # 向下翻页, 9 表示翻 9 页
    9Ctrl+B                         # 向上翻页, 9 表示翻 9 页  
       
}
